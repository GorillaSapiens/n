before call:
push stack pointer onto HARDWARE stack
push room for return onto ARGUMENT stack
push arguments onto ARGUMENT stack
jsr

function prologue:
;push frame pointer onto HARDWARE stack
lda fp+1
pha
lda fp
pha
;transfer stack pointer to frame pointer
lda sp+1
sta fp+1
lda sp
sta sp

function epilogue:
;pop frame pointer
pla
sta fp
pla
sta fp+1
;pop stack pointer
pla
sta sp
pla
sta sp+1
rts
